SQL> CREATE OR REPLACE TRIGGER student_total_per
BEFORE INSERT ON student
FOR EACH ROW
BEGIN
SET NEW.total = NEW.sub1 + NEW.sub2 + NEW.sub3, NEW.per = NEW.total / 3;
END;
/

SQL> CREATE OR REPLACE TRIGGER student_update
BEFORE INSERT ON student
FOR EACH ROW
BEGIN
    IF (sub1==NEW.sub1)
SET NEW.total = NEW.sub1 + NEW.sub2 + NEW.sub3, NEW.per = NEW.total / 3;
END;
/
